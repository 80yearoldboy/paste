// 手动设置会话 const accessToken = "您的访问令牌"; const refreshToken = "您的刷新令牌"; const supabaseUrl = "您的Supabase URL"; const supabaseKey = "您的Supabase Key"; // 直接调用Supabase API设置会话 fetch(`${supabaseUrl}/auth/v1/user`, { method: 'GET', headers: { 'Authorization': `Bearer ${accessToken}`, 'apikey': supabaseKey } }) .then(response => response.json()) .then(user => { console.log("用户信息:", user); // 保存用户信息到本地存储 localStorage.setItem('supabase_user', JSON.stringify(user)); localStorage.setItem('supabase_access_token', accessToken); localStorage.setItem('supabase_refresh_token', refreshToken); alert("验证成功！"); window.location.href = "dashboard.html"; }) .catch(error => { console.error("验证失败:", error); alert("验证失败: " + error.message); });
